            List<int> liczby = new List<int>() { 5, 4, 1, 3, 9, 8, 6, 7, 2, 0 };
            // Składnia zapytania
            IEnumerable<int> składniaZapytania =
                from liczba in liczby
                where liczba < 3 || liczba > 7
                select liczba;
            // Składnia metody
            IEnumerable<int> składniaMetody = liczby.Where(liczba => liczba < 3 || liczba > 7);
            foreach (var liczba in składniaZapytania)
                Console.Write(liczba + " ");
            Console.WriteLine();
            foreach (var liczba in składniaMetody)
                Console.Write(liczba + " ");
            // Wynik: 
            // "1 9 8 2 0 "
