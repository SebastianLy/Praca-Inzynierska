            List<int> liczby = new List<int>() { 5, 4, 1, 3, 9, 8, 6, 7, 2, 0 };
            // Składnia zapytań
            IEnumerable<int> składniaZapytań = 
                from liczba in liczby
                where liczba < 3 || liczba > 7 
                select liczba;
            // Składnia metod
            IEnumerable<int> składniaMetod = liczby.Where(liczba => liczba < 3 || liczba > 7);
            foreach (var liczba in składniaZapytań)
                Console.Write(liczba + " ");
            Console.WriteLine();
            foreach (var liczba in składniaMetod)
                Console.Write(liczba + " ");
            // Wynik: 
            // "1 9 8 2 0 "
